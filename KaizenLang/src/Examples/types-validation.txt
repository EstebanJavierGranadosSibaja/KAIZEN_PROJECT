// Types validation for KaizenLang

// 1) Primitive types
integer a = 10;
float b = 3.1415;
string s = "texto";
bool t = true;
// EXPECT ERROR: 'null' is a literal, not a type; this line is intentionally invalid
null n = null;

// 2) Implicit conversion attempt (should be rejected by strict typing)
integer bad = "123"; // semantically invalid
// EXPECT ERROR: assigning string literal to integer should be rejected by semantic analyzer

// 3) Assign input conversions (runtime may convert)
integer from_input = input();
string from_input_s = input();
// Note: input() builtin returns string tokens; runtime may convert to declared type or produce error

// 4) Arrays (single-dimension)
array<integer> nums = [1, 2, 3, 4];
array<string> names = ["Ana", "Luis", "María"];

// 5) Matrices (array of arrays)
matrix<integer> mat = [[1,2,3],[4,5,6],[7,8,9]];

// 6) Type mismatch in arrays (should be rejected)
array<integer> mixed = [1, "dos", 3];
// EXPECT ERROR: element type mismatch inside array literal

// 7) Operations preserving types
integer sum = nums[0] + nums[1];
float scaled = b * 2.0;
string greet = s + " Kaizen";

// 8) Null assignment checks
integer maybe = null; // EXPECT ERROR: assigning null to non-nullable primitive

// 9) Edge cases: empty arrays
array<integer> empty = [];

// 10) Access out of bounds (interpreter runtime error expected)
integer oob = nums[10];

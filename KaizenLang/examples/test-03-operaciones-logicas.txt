output("=== OPERACIONES DE COMPARACIÓN ===");
gear a = 10;
gear b = 20;
gear c = 10;

shin mayorQue = a > b;
shin menorQue = a < b;
shin mayorIgual = a >= c;
shin menorIgual = b <= 100;
shin igual = a == c;
shin diferente = a != b;

output("a = " + a + ", b = " + b + ", c = " + c);
output("a > b: " + mayorQue);
output("a < b: " + menorQue);
output("a >= c: " + mayorIgual);
output("b <= 100: " + menorIgual);
output("a == c: " + igual);
output("a != b: " + diferente);

output("=== OPERACIONES LÓGICAS (AND) ===");
shin verdadero = true;
shin falso = false;

shin and1 = verdadero && verdadero;
shin and2 = verdadero && falso;
shin and3 = falso && falso;

output("true && true: " + and1);
output("true && false: " + and2);
output("false && false: " + and3);

output("=== OPERACIONES LÓGICAS (OR) ===");
shin or1 = verdadero || verdadero;
shin or2 = verdadero || falso;
shin or3 = falso || falso;

output("true || true: " + or1);
output("true || false: " + or2);
output("false || false: " + or3);

output("=== OPERACIONES LÓGICAS (NOT) ===");
shin not1 = !verdadero;
shin not2 = !falso;

output("!true: " + not1);
output("!false: " + not2);

output("=== EXPRESIONES LÓGICAS COMPLEJAS ===");
gear edad = 25;
shin tieneCarnet = true;

shin puedeConducir = (edad >= 18) && tieneCarnet;
output("Edad >= 18 y tiene carnet: " + puedeConducir);

gear nota = 85;
shin aprobado = (nota >= 70) || (nota == 100);
output("Nota >= 70 o nota == 100: " + aprobado);

shin complejo = ((a > 5) && (b < 30)) || (c == 10);
output("((a > 5) && (b < 30)) || (c == 10): " + complejo);

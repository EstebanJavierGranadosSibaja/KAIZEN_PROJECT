void saludar() ying

output("=== FUNCIONES VOID CON PARÁMETROS ===");

void saludarPersona(string nombre) ying
    output("¡Hola " + nombre + "!");
yang

saludarPersona("Ana");
saludarPersona("Luis");

void mostrarNumero(integer num) ying
    output("El número es: " + num);
yang

mostrarNumero(42);
mostrarNumero(100);

output("=== FUNCIONES CON RETORNO ===");

integer suma(integer a, integer b) ying
    return a + b;
yang

integer resultado = suma(10, 5);
output("Suma de 10 + 5: " + resultado);

integer resultado2 = suma(25, 75);
output("Suma de 25 + 75: " + resultado2);

output("=== FUNCIÓN DE MULTIPLICACIÓN ===");

integer multiplicar(integer x, integer y) ying
    return x * y;
yang

integer mult1 = multiplicar(5, 6);
integer mult2 = multiplicar(12, 3);

output("5 * 6 = " + mult1);
output("12 * 3 = " + mult2);

output("=== FUNCIÓN CON FLOAT ===");

float dividir(float a, float b) ying
    return a / b;
yang

float div1 = dividir(10.0, 2.0);
float div2 = dividir(7.5, 2.5);

output("10.0 / 2.0 = " + div1);
output("7.5 / 2.5 = " + div2);

output("=== FUNCIÓN CON BOOL ===");

bool esMayor(integer a, integer b) ying
    return a > b;
yang

bool comp1 = esMayor(10, 5);
bool comp2 = esMayor(3, 8);

output("10 > 5: " + comp1);
output("3 > 8: " + comp2);

output("=== FUNCIÓN CON STRING ===");

string concatenar(string a, string b) ying
    return a + " " + b;
yang

string frase = concatenar("Hola", "Mundo");
output("Concatenación: " + frase);

output("=== FUNCIÓN CON MÚLTIPLES PARÁMETROS ===");

integer calcular(integer a, integer b, integer c) ying
    return (a + b) * c;
yang

integer calc1 = calcular(5, 3, 2);
output("(5 + 3) * 2 = " + calc1);

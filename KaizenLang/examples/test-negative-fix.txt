// Test para verificar que los números negativos funcionan correctamente
// También prueba precisión de bankai
// Enteros negativos
gear a = -10;
gear b = -1000000;

// Flotantes negativos
bankai c = -2.5;
bankai d = -3.141592653589793;

// Número científico grande
bankai e = 299792458.0;

// Operaciones con negativos
gear suma = a + 5;           // -10 + 5 = -5
gear resta = a - b;          // -10 - (-1000000) = 999990
bankai mult = c * 2.0;          // -2.5 * 2.0 = -5.0

// Doble negación
gear f = -(-10);             // -(-10) = 10

// Expresiones complejas
gear g = -(5 + 3);           // -(8) = -8

output("a =", a);               // Debe ser -10
output("b =", b);               // Debe ser -1000000
output("c =", c);               // Debe ser -2.5
output("d =", d);               // Debe ser -3.141592653589793 (precisión completa)
output("e =", e);               // Debe ser 299792458.0 (sin redondeo)
output("suma =", suma);         // Debe ser -5
output("resta =", resta);       // Debe ser 999990
output("mult =", mult);         // Debe ser -5.0
output("f =", f);               // Debe ser 10
output("g =", g);               // Debe ser -8

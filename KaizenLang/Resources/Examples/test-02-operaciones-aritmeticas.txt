// Prueba todas las operaciones aritméticas básicas

output("=== OPERACIONES CON INTEGERS ===");
integer a = 10;
integer b = 3;

integer suma = a + b;
integer resta = a - b;
integer multiplicacion = a * b;
integer division = a / b;

output("a = " + a + ", b = " + b);
output("Suma: " + a + " + " + b + " = " + suma);
output("Resta: " + a + " - " + b + " = " + resta);
output("Multiplicación: " + a + " * " + b + " = " + multiplicacion);
output("División: " + a + " / " + b + " = " + division);

output("=== OPERACIONES CON FLOATS ===");
float x = 10.5;
float y = 2.5;

float sumaF = x + y;
float restaF = x - y;
float multF = x * y;
float divF = x / y;

output("x = " + x + ", y = " + y);
output("Suma: " + x + " + " + y + " = " + sumaF);
output("Resta: " + x + " - " + y + " = " + restaF);
output("Multiplicación: " + x + " * " + y + " = " + multF);
output("División: " + x + " / " + y + " = " + divF);

output("=== OPERACIONES MIXTAS (PROMOCIÓN) ===");
integer entero = 5;
float decimal = 2.5;
float resultado = entero + decimal;
output("Integer + Float: " + entero + " + " + decimal + " = " + resultado);

output("=== EXPRESIONES COMPLEJAS ===");
integer complejo = (10 + 5) * 2 - 8 / 4;
output("(10 + 5) * 2 - 8 / 4 = " + complejo);

float complejo2 = 10.0 / 3.0 + 5.5 * 2.0;
output("10.0 / 3.0 + 5.5 * 2.0 = " + complejo2);

output("=== NÚMEROS NEGATIVOS ===");
integer neg1 = -15;
integer neg2 = -5;
integer sumaNeg = neg1 + neg2;
integer restaNeg = neg1 - neg2;
output("Negativos: " + neg1 + " + " + neg2 + " = " + sumaNeg);
output("Negativos: " + neg1 + " - " + neg2 + " = " + restaNeg);
